 SELECT I.ID, I.SUMMARY, I.DESCRIPTION, IL.NAME
        FROM ISSUES I
        JOIN ISSUESLIST IL ON I.ISSUESLIST_ID= IL.ID #16.4.1
        
        SELECT I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME, U.LASTNAME FROM ISSUES I
        JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
        
        SELECT * FROM ISSUES
        UPDATE ISSUES 
        SET USER_ID_ASSIGNEDTO=1
        WHERE ID=4
        
        SELECT FIRSTNAME, LASTNAME, COUNT(*) AS TASK_NUMBER FROM USERS U
		JOIN ISSUES I ON I.USER_ID_ASSIGNEDTO = U.ID
        GROUP BY I.USER_ID_ASSIGNEDTO
        HAVING COUNT(*) > 1
